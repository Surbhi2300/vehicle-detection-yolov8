# YOLOv8 Commands Used in Project

# Training
yolo detect train model=yolov8n.pt data=data/roboflow_dataset/data.yaml epochs=50 imgsz=640

# Image inference
yolo detect predict model=models/best.pt source=data/raw_images/test.png conf=0.25 show=True

# Video inference
yolo detect predict model=models/best.pt source=data/raw_videos/traffic_short.mp4 conf=0.15 show=True
